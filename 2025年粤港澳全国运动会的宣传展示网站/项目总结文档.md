## 一、项目概述

### 1.1 项目背景

本项目是为2025年第十五届全国运动会（粤港澳赛区）开发的官方宣传展示网站。粤港澳三地首次联合举办全运会，开创了跨境办赛的先河，具有历史性意义。本网站旨在全面展示赛事信息、运动员风采、新闻动态等内容，为公众提供便捷的信息获取渠道。

### 1.2 项目目标

- 建设一个功能完善、界面美观的运动会宣传网站
- 实现多页面联动与统一视觉风格
- 提供良好的用户体验（响应式设计、交互动画）
- 展示现代Web开发技术（CSS3动画、JavaScript高级应用）

```html
<section class="hero-slider">
    <div class="slider-container">

        <input type="radio" name="slider" id="slide1" checked>
        <input type="radio" name="slider" id="slide2">
        <input type="radio" name="slider" id="slide3">

        <!-- 轮播内容 -->

            <div class="slide">...</div>
            <div class="slide">...</div>
            <div class="slide">...</div>
        </div>

        <!-- 指示点 -->
        <div class="slider-dots">
            <label for="slide1" class="dot"></label>
            <label for="slide2" class="dot"></label>
            <label for="slide3" class="dot"></label>
        </div>
    </div>
</section>
```

**CSS控制显示**：
```css
#slide1:checked ~ .slides .slide:nth-child(1),
#slide2:checked ~ .slides .slide:nth-child(2),
#slide3:checked ~ .slides .slide:nth-child(3) {
    opacity: 1;
    z-index: 1;
}

.slide {
    position: absolute;
    top: 0;
    left: 0;
    width: 100%;
    height: 100%;
    opacity: 0;
    transition: opacity 0.8s ease-in-out;
}
```

**JavaScript自动轮播**：
```javascript
carousel() {
    var slider = document.querySelector('.hero-slider');
    if (slider) {
        var slides = slider.querySelectorAll('.slide');
        var cur = 0;
        var total = slides.length;

        // 5秒自动切换
        setInterval(() => {
            cur = (cur + 1) % total;
            var r = slider.querySelector('#slide' + (cur + 1));
            if (r) r.checked = true;
        }, 5000);
    }
}
```

##### 功能4：元素进入视口动画

**实现方法**：IntersectionObserver API + data-animate属性

**HTML**：
```html
<div class="overview-card" data-animate="fade-in">
    <div class="card-icon">🏆</div>
    <h3>参赛规模</h3>
    <p>...</p>
</div>
```

**CSS**：
```css
[data-animate] {
    opacity: 0;
    transform: translateY(30px);
    transition: all 0.6s ease;
}

[data-animate].animated {
    opacity: 1;
    transform: translateY(0);
}
```

**JavaScript**：
```javascript
anim() {
    var els = document.querySelectorAll('[data-animate]');
    var obs = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) entry.target.classList.add('animated');
        });
    }, { threshold: 0.1 });
    els.forEach(el => obs.observe(el));
}
```

```javascript
formCheck() {
    var forms = document.querySelectorAll('form');
    forms.forEach(form => {
        form.addEventListener('submit', (e) => {
            if (!this.checkForm(form)) e.preventDefault();
        });
        var inputs = form.querySelectorAll('input, textarea, select');
        inputs.forEach(inp => {
            inp.addEventListener('blur', () => this.checkField(inp));
            inp.addEventListener('input', () => this.clearErr(inp));
        });
    });
}

checkForm(form) {
    var ok = true;
    var inputs = form.querySelectorAll('input[required], textarea[required], select[required]');
    inputs.forEach(inp => { if (!this.checkField(inp)) ok = false; });
    return ok;
}

checkField(f) {
    var v = f.value.trim();
    var t = f.type;
    var ok = true;
    var msg = '';

    // 必填检查
    if (f.hasAttribute('required') && !v) {
        ok = false;
        msg = '必填';
    }

    // 邮箱格式检查
    if (t === 'email' && v) {
        var re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        if (!re.test(v)) { ok = false; msg = '邮箱格式不对'; }
    }

    // 手机号格式检查（中国手机号）
    if (t === 'tel' && v) {
        var re2 = /^1[3-9]\d{9}$/;
        if (!re2.test(v)) { ok = false; msg = '手机格式不对'; }
    }

    this.showErr(f, ok, msg);
    return ok;
}

showErr(f, ok, msg) {
    var g = f.closest('.form-group');
    if (!g) return;
    g.classList.remove('error', 'success');

    if (!ok) {
        g.classList.add('error');
        var e = g.querySelector('.error-message');
        if (!e) {
            e = document.createElement('span');
            e.className = 'error-message';
            g.appendChild(e);
        }
        e.textContent = msg;
    } else if (f.value.trim()) {
        g.classList.add('success');
    }
}

clearErr(f) {
    var g = f.closest('.form-group');
    if (g) {
        g.classList.remove('error');
        var e = g.querySelector('.error-message');
        if (e) e.remove();
    }
}
```

## 四、核心功能实现原理

### 4.1 纯CSS筛选功能原理

**技术名称**：Checkbox Hack（复选框黑客技术）

**原理说明**：利用HTML中`<label>`标签可以关联`<input type="radio">`的特性，通过CSS的`:checked`伪类选择器和相邻兄弟选择器（`~`）来控制页面元素的显示和隐藏，从而实现无需JavaScript的交互功能。

**完整实现步骤**：

1. **HTML结构**：放置隐藏的radio input元素
2. **交互元素**：使用label作为可点击的按钮
3. **目标元素**：需要控制显示/隐藏的内容
4. **CSS选择器**：使用`input:checked ~ .container .target`来匹配

**代码示例**：

```html
<!-- 1. 隐藏的radio按钮 -->
<input type="radio" name="filter" id="filter-all" checked>
<input type="radio" name="filter" id="filter-a">
<input type="radio" name="filter" id="filter-b">

<!-- 2. 交互按钮（label） -->
<div class="buttons">
    <label for="filter-all">全部</label>
    <label for="filter-a">A类</label>
    <label for="filter-b">B类</label>
</div>

<!-- 3. 目标元素 -->
<div class="items">
    <div class="item" data-category="a">A-1</div>
    <div class="item" data-category="a">A-2</div>
    <div class="item" data-category="b">B-1</div>
</div>
```

```css
/* 隐藏radio */
input[type="radio"] { display: none; }

/* 默认显示所有 */
.item { display: none; }
#filter-all:checked ~ .items .item { display: block; }

/* 筛选显示 */
#filter-a:checked ~ .items .item[data-category="a"] { display: block; }
#filter-b:checked ~ .items .item[data-category="b"] { display: block; }

/* 按钮选中状态 */
#filter-all:checked ~ .buttons label[for="filter-all"],
#filter-a:checked ~ .buttons label[for="filter-a"],
#filter-b:checked ~ .buttons label[for="filter-b"] {
    background: blue;
    color: white;
}
```

### 4.2 轮播图实现原理

**方案**：CSS Radio + JavaScript定时器

**原理说明**：

### 4.3 图片懒加载实现原理

**技术**：IntersectionObserver API

**原理说明**：
- 传统的scroll事件监听方式性能较差
- IntersectionObserver是浏览器原生提供的API，可以高效监听元素是否进入视口
- 当元素进入视口时，将data-src的值赋给src属性，开始加载图片
- 加载完成后停止观察该元素

**代码示例**：

```javascript
lazyImg() {
    var imgs = document.querySelectorAll('img[data-src]');
    var obs = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                var img = entry.target;
                img.src = img.dataset.src;  // 开始加载图片
                img.classList.remove('lazy');
                img.classList.add('loaded');
                obs.unobserve(img);  // 停止观察
            }
        });
    });
    imgs.forEach(img => obs.observe(img));
}
```

### 4.5 表单验证实现原理

**技术**：HTML5原生验证 + JavaScript自定义验证

**验证流程**：

```
┌─────────────┐    ┌─────────────┐    ┌─────────────┐
│  用户输入   │───▶│  失去焦点   │───▶│  验证函数   │
│             │    │  (blur)     │    │  执行       │
└─────────────┘    └─────────────┘    └──────┬──────┘
                                             │
                    ┌────────────────────────┘
                    ▼
           ┌────────────────┐    ┌────────────────┐
           │  验证通过      │───▶│  显示成功样式  │
           └────────────────┘    └────────────────┘
                    │
                    ▼
           ┌────────────────┐    ┌────────────────┐
           │  验证失败      │───▶│  显示错误信息  │
           └────────────────┘    └────────────────┘
```

**代码示例**：

```javascript
checkField(f) {
    var v = f.value.trim();
    var t = f.type;
    var ok = true;
    var msg = '';

    // 1. 必填验证
    if (f.hasAttribute('required') && !v) {
        ok = false;
        msg = '必填';
    }

    // 2. 邮箱格式验证（正则表达式）
    if (t === 'email' && v) {
        var re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        if (!re.test(v)) { ok = false; msg = '邮箱格式不对'; }
    }

    // 3. 手机号格式验证
    if (t === 'tel' && v) {
        var re2 = /^1[3-9]\d{9}$/;
        if (!re2.test(v)) { ok = false; msg = '手机格式不对'; }
    }

    // 4. 显示结果
    this.showErr(f, ok, msg);
    return ok;
}
```

```javascript
gesture() {
    var els = document.querySelectorAll('.gesture-area');
    var sx, sy, st;

    els.forEach(el => {
        // 触摸开始
        el.addEventListener('touchstart', (e) => {
            sx = e.touches[0].clientX;
            sy = e.touches[0].clientY;
            st = Date.now();
        });

        // 触摸结束
        el.addEventListener('touchend', (e) => {
            if (!sx || !sy) return;

            var touch = e.changedTouches[0];
            var dx = sx - touch.clientX;
            var dy = sy - touch.clientY;
            var dt = Date.now() - st;

            // 判断滑动方向（水平或垂直）
            if (Math.abs(dx) > Math.abs(dy)) {
                // 水平滑动
                if (Math.abs(dx) > 50 && dt < 300) {
                    this.swipe(el, dx > 0 ? 'left' : 'right');
                }
            } else {
                // 垂直滑动
                if (Math.abs(dy) > 50 && dt < 300) {
                    this.swipe(el, dy > 0 ? 'up' : 'down');
                }
            }

            sx = sy = null;
        });
    });
}
```
